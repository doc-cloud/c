#include "dlinklist.h"

#include <stdlib.h>

typedef struct {
	dlinklistnode header;
	dlinklistnode *slider;
	int length;
} dlinklist_t;

dlinklist *dlinklist_create(void)
{
	dlinklist_t *dlist = (dlinklist_t *)malloc(sizeof(dlinklist_t));
	if (NULL != dlist) {
		dlist->header.next = NULL;
		dlist->header.prev = NULL;
		dlist->slider = NULL;
		dlist->length = 0;
	}
	return dlist;
}
